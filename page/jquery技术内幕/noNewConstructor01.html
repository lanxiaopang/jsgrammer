虽然构造函数有默认方法，但是返回的方法（return中的sayName()）先！！！！！

    function p(){
        this.name = name;
        this.sayName = function(){
            alert(this.name);
        };
        return {
            "name":"name",
            sayName : function(){
                alert("1!!!");
            }
        };
    };
    var p = new p("高寻");
    p.sayName();
